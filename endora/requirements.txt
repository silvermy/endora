# Endora — Python dependencies
# Built into the Docker image at build time.

# ── Computer vision ───────────────────────────────────────────────────────────
# Ultralytics YOLO: body pose estimation (replaces MediaPipe Pose)
# Official arm64/amd64 wheels; nano pose model (~7 MB) is downloaded at build time.
ultralytics>=8.3.0

# MediaPipe: required by grlib for hand landmark detection
# Official arm64 wheels exist from v0.10.9 (works on Pi 4/5 64-bit, amd64)
mediapipe>=0.10.9

# grlib runtime deps (grlib itself is installed --no-deps in the Dockerfile to
# avoid its numpy~=1.21.4 pin conflicting with ultralytics>=1.23.0 requirement).
pandas>=1.1.4
imutils>=0.5.4

# ONNX Runtime — used instead of PyTorch for YOLO inference on ARM64.
# The standard PyTorch conv2d kernel uses ARMv8.2-A dot-product instructions
# absent on Cortex-A72 (Pi 4), causing SIGILL.  ONNX Runtime is compiled
# conservatively and works on all ARMv8-A CPUs.
onnxruntime>=1.16.0

# OpenCV headless — no Qt/GTK, smaller image, all we need is VideoCapture
# Must be built with FFmpeg support for RTSP (the PyPI wheel includes it)
opencv-python-headless>=4.9.0.80

# ── Config ────────────────────────────────────────────────────────────────────
PyYAML>=6.0

# ── Notes ─────────────────────────────────────────────────────────────────────
# Output to HA uses urllib (stdlib) — no extra packages needed.
# MQTT and HTTP backends have been removed; HA Supervisor API is the only
# output target.  If you want MQTT as well, add: paho-mqtt>=2.0.0
